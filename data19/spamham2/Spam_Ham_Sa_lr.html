    ## 1. 전처리 1번: 글자가 아닌 것은 공백으로 대체
    pre1 = re.sub(r"[^a-zA-Z]+", " ", str(v2))

    ## 2. 소문자로 변경하여 단어를 토큰으로 마들자.
    pre1_lower_token = word_tokenize(pre1.lower())
    # print(pre1_lower_token)
    
    ## 3. 전처리 2번: 불용어 처리
    pre2 = [w for w in pre1_lower_token if w not in stopwords]
    # print(pre2)

    # print(pre3)
    words.append(pre3)

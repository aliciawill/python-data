# 단어들만이 아닌 문장을 입력해도 워드클라우드로 생성된다
text = '티셔츠 · 블라우스/셔츠 · 니트/스웨터 · 조끼/베스트 · 가디건 · 자켓/코트/점퍼 · 원피스 · 청바지/진' # 워드클라우드로 생성할 내용
# Generate word cloud 
wordcloud = WordCloud(font_path='../data04/NanumGothic.ttf').generate(text)
# Set figure size
plt.figure(figsize=(10, 10))
# Display image
plt.imshow(wordcloud) 
# No axis details
plt.axis("off")
plt.show()
